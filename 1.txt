class Cat
{
private:
    string _name;
    int _age;
    string _color;
    bool _male;

public:
    string GetName()
    {
        return _name;
    }

    void SetName(string name)
    {
        _name = name;
    }

    int GetAge()
    {
        return _age;
    }

    void SetAge(int age)
    {
        _age = age;
    }

    string GetColor()
    {
        return _color;
    }

    void SetColor(string color)
    {
        _color = color;
    }

    bool GetPol()
    {
        return _male;
    }

    void SetPol(bool male)
    {
        _male = male;
    }

    Cat(string name = "Барсик", int age = 3, string color = "Полосатый", bool male = true)
    {
        _name = name;
        _age = age;
        _color = color;
        _male = male;
    }

    Cat(const Cat& cat)
    {
        _name = cat._name;
        _age = cat._age;
        _color = cat._color;
        _male = cat._male;
    }
    
    ~Cat()
    {
        cout << "Имя кота deleted";
    }
};
